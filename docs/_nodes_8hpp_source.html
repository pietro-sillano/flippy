71 void pop_nn(Index to_pop_nn_id)
72 {
80 auto pop_pos = find_nns_loc_pointer(to_pop_nn_id);
81 auto dist = pop_pos - nn_ids.begin();
82
83 if (pop_pos!=nn_ids.end()) {
84 // I checked that this would work on example code on cppreference https://godbolt.org/z/6qf8c9nTz
85 nn_ids.erase(pop_pos);
86 nn_distances.erase(nn_distances.begin() + dist);
87 }

108 // unit-tested
109 void emplace_nn_id(Index to_emplace_nn_id, vec3<Real> const& to_emplace_nn_pos, Index loc_idx)
110 {
126 if (loc_idx<nn_ids.size()) {
127 auto signed_loc_idx = static_cast<long long >(loc_idx);
128 nn_ids.emplace(nn_ids.begin() + signed_loc_idx, to_emplace_nn_id);
129 nn_distances.emplace(nn_distances.begin() + signed_loc_idx, to_emplace_nn_pos - pos);
130 }
131 }

133 //unit-tested
135 vec3<Real> const& get_distance_vector_to(Index nn_id) const
136 {
144 auto id_pos = std::find(nn_ids.begin(), nn_ids.end(), nn_id);
145 if (id_pos!=nn_ids.end()) {
146 return nn_distances[static_cast<Index>(id_pos - nn_ids.begin())];
147 }
148 else {
149 std::cerr << "nn_id:" << nn_id << " provided to `get_distance_vector_to` is not a next neighbour of the node "
150 << id;
151 exit(12);
152 }
