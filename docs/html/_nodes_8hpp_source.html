61 void pop_nn(Index to_pop_nn_id)
62 {
63 // finds the element with the id to_pop_nn_id in the nn_id vector and deletes it.
64 // this will lead to resizing of the vector!
65 auto pop_pos = find_nns_loc_pointer(to_pop_nn_id);
66 Index dist = pop_pos - nn_ids.begin();
67
68 if (pop_pos!=nn_ids.end()) {
69 // I checked that this would work on example code on cppreference https://godbolt.org/z/6qf8c9nTz
70 nn_ids.erase(pop_pos);
71 nn_distances.erase(nn_distances.begin() + dist);
72 }
73 }
