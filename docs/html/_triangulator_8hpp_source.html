157 static TriangleRegion get_region(Index i, Index j, Index sizeMinOne)
158 {
159 if (i==0) { return TOP_CORNER; }
160 else if (j==0 && i==sizeMinOne) { return BOTTOM_LEFT_CORNER; }
161 else if (j==sizeMinOne && i==sizeMinOne) { return BOTTOM_RIGHT_CORNER; }
162 else if (j==0) { return LEFT_EDGE; }
163 else if (i==sizeMinOne) { return BOTTOM_EDGE; }
164 else if (i==j) { return DIAGONAL_EDGE; }
165 else { return BULK; }

240 static Real even_angular_distance_length(Real l, Index k, Index n, Real R = 1.)
241 {
246 if (k==0) {
247 return 0;
248 }
249 else {
250 Real fr = static_cast<Real>(k)/static_cast<Real>(n);
251 Real denominator = l + sqrt(4.*R*R - l*l)/tan(fr*2.*asin(l/(2.*R)));
252 return 2.*R*R/denominator;
253 }

256 static vec3<Real> get_pos(vec3<Real> const& p0, vec3<Real> const& p1, vec3<Real> const& p2, Index i, Index j, Index maxIdx)
257 {
269 vec3<Real> e1 = p1 - p0;
270 vec3<Real> e2 = p2 - p0;
271 auto e = e1.norm();
272 Real wi = even_angular_distance_length(e, i, maxIdx);
273
274 e1.normalize();
275 e2.normalize();
276
277 vec3<Real> li{};
278 Real li_norm{0};
279 vec3<Real> interm_1 = p0 + wi*e1;
280 interm_1.normalize();
281 if (i!=0) {
282 vec3<Real> interm_2 = p0 + wi*e2;
283 interm_2.normalize();
284
285 li = interm_2 - interm_1;
286 li_norm = li.norm();
287 li.normalize();
288 }
289 Real wj = even_angular_distance_length(li_norm, j, i);
290 return interm_1 + wj*li;
291 }
