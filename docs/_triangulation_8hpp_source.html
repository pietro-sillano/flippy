594 void update_bulk_node_geometry(Index node_id)
595 {
596 update_nn_distance_vectors(node_id);
597
598 Real area_sum = 0.;
599 vec3<Real> face_normal_sum{0., 0., 0.}, local_curvature_vec{0., 0., 0.};
600 vec3<Real> face_normal;
601 indexing_number auto nn_number = (Index) nodes_.nn_ids(node_id).size();
602 Index j_p_1;
603
604 Real face_area, face_normal_norm;
605 vec3<Real> ljj_p_1, lij_p_1, lij;
606 Real cot_at_j, cot_at_j_p_1;
607
608 for (Index j = 0; j<nn_number; ++j) {
609 //return j+1 element of ordered_nn_ids unless j has the last value then wrap around and return 0th element
610 j_p_1 = Neighbors<Index>::plus_one(j,nn_number);
611
612 lij = nodes_.nn_distances(node_id)[j];
613 lij_p_1 = nodes_.nn_distances(node_id)[j_p_1];
614 ljj_p_1 = lij_p_1 - lij;
615
616 cot_at_j = cot_between_vectors(lij, (-1)*ljj_p_1);
617 cot_at_j_p_1 = cot_between_vectors(lij_p_1, ljj_p_1);
618
619
620 face_normal = lij.cross(lij_p_1); //nodes_.nn_distances(node_id)[j].cross(nodes_.nn_distances(node_id)[j_p_1]);
621 face_normal_norm = face_normal.norm();
622#ifdef DEBUG
623 if(face_normal_norm < 1e-10) {
624 throw std::runtime_error("A triangle face is degenerate and Area sum is evaluating to "+std::to_string(face_normal_norm)+". This should not happen.");
625 }
626#endif
627 face_area = mixed_area(lij, lij_p_1, Real(0.5)*face_normal_norm, cot_at_j, cot_at_j_p_1);
628 area_sum += face_area;
629 face_normal_sum += face_area*face_normal/face_normal_norm;
630
631 local_curvature_vec -= (cot_at_j_p_1*lij + cot_at_j*lij_p_1);
632 }
633 nodes_.set_area(node_id, area_sum);
634 nodes_.set_volume(node_id, nodes_[node_id].pos.dot(face_normal_sum)/((Real) 3.)); // 18=3*6: 6 has the aforementioned justification. 3 is part of the formula for the tetrahedron volume
635 nodes_.set_curvature_vec(node_id, -local_curvature_vec/((Real) 2.*area_sum)); // 2 is part of the formula to calculate the local curvature I just did not divide the vector inside the loop
636 nodes_.set_unit_bending_energy(node_id, local_curvature_vec.dot(local_curvature_vec)/((Real) 8.*area_sum)); // 8 is 2*4, where 4 is the square of the above two and the area in the denominator is what remains after canceling. 1/ comes from the pre-factor to bending energy
637
638 };

685 static Real mixed_area(vec3<Real> const& lij, vec3<Real> const& lij_p_1, Real triangle_area, Real cot_at_j, Real cot_at_j_p_1){
686 if ((cot_at_j>0.) && (cot_at_j_p_1>0.)) { // both angles at j and j+1 are smaller than 90 deg so the triangle can only be obtuse at the node
687 if (lij.dot(lij_p_1)>0) { // cos at i is positive i.e. angle at i is not obtuse
688 return (cot_at_j_p_1*lij.dot(lij) + cot_at_j*lij_p_1.dot(lij_p_1))/Real(8.);
689 }
690 else {//obtuse at node i.
691 return triangle_area/Real(2.);
692 }
693 }
694 else {//obtuse at node j or j+1.
695 return triangle_area/Real(4.);
696 }
697
698 }
