242 BondFlipData<Index> flip_bond(Index node_id, Index nn_id,
243 Real min_bond_length_square,
244 Real max_bond_length_square)
245 {
246 BondFlipData<Index> bfd{};
247 if (nodes_.nn_ids(node_id).size()>BOND_DONATION_CUTOFF) {
248 if (nodes_.nn_ids(nn_id).size()>BOND_DONATION_CUTOFF) {
249 Neighbors<Index> common_nns = previous_and_next_neighbour_global_ids(node_id, nn_id);
250 Real bond_length_square = (nodes_.pos(common_nns.j_m_1) - nodes_.pos(common_nns.j_p_1)).norm_square();
251 if ((bond_length_square<max_bond_length_square) && (bond_length_square>min_bond_length_square)) {
252 if (common_neighbours(node_id, nn_id).size()==2) {
253 pre_update_geometry = calculate_diamond_geometry(node_id, nn_id, common_nns.j_m_1, common_nns.j_p_1);
254 bfd = flip_bond_unchecked(node_id, nn_id, common_nns.j_m_1, common_nns.j_p_1);
255 if (common_neighbours(bfd.common_nn_0, bfd.common_nn_1).size()==2) {
256 update_diamond_geometry(node_id, nn_id, common_nns.j_m_1, common_nns.j_p_1);
257 post_update_geometry = calculate_diamond_geometry(node_id, nn_id, common_nns.j_m_1,
258 common_nns.j_p_1);
259 update_global_geometry(pre_update_geometry, post_update_geometry);
260 }
261 else {
262 flip_bond_unchecked(bfd.common_nn_0, bfd.common_nn_1, nn_id, node_id);
263 bfd.flipped = false;
264 }
265 }
266 }
267 }
268 }
269 return bfd;

290 // unit-tested
291 void update_bulk_node_geometry(Index node_id)
292 {
297 update_nn_distance_vectors(node_id);
298
299 Real area_sum = 0.;
300 vec3<Real> face_normal_sum{0., 0., 0.}, local_curvature_vec{0., 0., 0.};
301 vec3<Real> face_normal;
302 integer_number auto nn_number = (Index) nodes_.nn_ids(node_id).size();
303 Index j_p_1;
304
305 Real face_area, face_normal_norm;
306 vec3<Real> ljj_p_1, lij_p_1, lij;
307 Real cot_at_j, cot_at_j_p_1;
308
309 for (Index j = 0; j<nn_number; ++j) {
310 //return j+1 element of ordered_nn_ids unless j has the last value then wrap around and return 0th element
311 j_p_1 = Neighbors<Index>::plus_one(j,nn_number);
312
313 lij = nodes_.nn_distances(node_id)[j];
314 lij_p_1 = nodes_.nn_distances(node_id)[j_p_1];
315 ljj_p_1 = lij_p_1 - lij;
316
317 cot_at_j = cot_between_vectors(lij, (-1)*ljj_p_1);
318 cot_at_j_p_1 = cot_between_vectors(lij_p_1, ljj_p_1);
319
320
321 face_normal = lij.cross(lij_p_1); //nodes_.nn_distances(node_id)[j].cross(nodes_.nn_distances(node_id)[j_p_1]);
322 face_normal_norm = face_normal.norm();
323 face_area = mixed_area(lij, lij_p_1, 0.5*face_normal_norm, cot_at_j, cot_at_j_p_1);
324 area_sum += face_area;
325 face_normal_sum += face_area*face_normal/face_normal_norm;
326
327 local_curvature_vec -= (cot_at_j_p_1*lij + cot_at_j*lij_p_1);
328 }
329
330 nodes_.set_area(node_id, area_sum);
331 nodes_.set_volume(node_id, nodes_[node_id].pos.dot(face_normal_sum)/((Real) 3.)); // 18=3*6: 6 has the aforementioned justification. 3 is part of the formula for the tetrahedron volume
332 nodes_.set_curvature_vec(node_id, -local_curvature_vec/((Real) 2.*area_sum)); // 2 is part of the formula to calculate the local curvature I just did not divide the vector inside the loop
333 nodes_.set_unit_bending_energy(node_id, local_curvature_vec.dot(local_curvature_vec)/((Real) 8.*area_sum)); // 8 is 2*4, where 4 is the square of the above two and the area in the denominator is what remains after canceling. 1/ comes from the pre-factor to bending energy
334
335 };

338 static std::tuple<Real, vec3<Real>> partial_voronoi_area_and_face_normal_of_node_in_a_triangle(vec3<Real> const& lij,
339 vec3<Real> const& lij_p_1)

356 static Real mixed_area(vec3<Real> const& lij, vec3<Real> const& lij_p_1, Real triangle_area, Real cot_at_j, Real cot_at_j_p_1){
357 if ((cot_at_j>0.) && (cot_at_j_p_1>0.)) { // both angles at j and j+1 are smaller than 90 deg so the triangle can only be obtuse at the node
358 if (lij.dot(lij_p_1)>0) { // cos at i is positive i.e. angle at i is not obtuse
359 return (cot_at_j_p_1*lij.dot(lij) + cot_at_j*lij_p_1.dot(lij_p_1))/8.;
360 }
361 else {//obtuse at node i.
362 return triangle_area/2.;
363 }
364 }
365 else {//obtuse at node j or j+1.
366 return triangle_area/4.;
367 }
368
369 }
