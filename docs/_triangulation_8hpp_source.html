376 Real min_bond_length_square,
377 Real max_bond_length_square){
399 BondFlipData<Index> bfd{};
400 if (nodes_.nn_ids(node_id).size()>BOND_DONATION_CUTOFF) {
401 if (nodes_.nn_ids(nn_id).size()>BOND_DONATION_CUTOFF) {
402 Neighbors<Index> common_nns = previous_and_next_neighbour_global_ids(node_id, nn_id);
403 Real bond_length_square = (nodes_.pos(common_nns.j_m_1) - nodes_.pos(common_nns.j_p_1)).norm_square();
404 if ((bond_length_square<max_bond_length_square) && (bond_length_square>min_bond_length_square)) {
405 if (common_neighbours(node_id, nn_id).size()==2) {
406 pre_update_geometry = calculate_diamond_geometry(node_id, nn_id, common_nns.j_m_1, common_nns.j_p_1);
407 bfd = flip_bond_unchecked(node_id, nn_id, common_nns.j_m_1, common_nns.j_p_1);
408 if (common_neighbours(bfd.common_nn_0, bfd.common_nn_1).size()==2) {
409 update_diamond_geometry(node_id, nn_id, common_nns.j_m_1, common_nns.j_p_1);
410 post_update_geometry = calculate_diamond_geometry(node_id, nn_id, common_nns.j_m_1,
411 common_nns.j_p_1);
412 update_global_geometry(pre_update_geometry, post_update_geometry);
413 }
414 else {
415 flip_bond_unchecked(bfd.common_nn_0, bfd.common_nn_1, nn_id, node_id);
416 bfd.flipped = false;
417 }
418 }
419 }
420 }
421 }

446 // unit-tested
447 void update_bulk_node_geometry(Index node_id)
448 {
453 update_nn_distance_vectors(node_id);
454
455 Real area_sum = 0.;
456 vec3<Real> face_normal_sum{0., 0., 0.}, local_curvature_vec{0., 0., 0.};
457 vec3<Real> face_normal;
458 indexing_number auto nn_number = (Index) nodes_.nn_ids(node_id).size();
459 Index j_p_1;
460
461 Real face_area, face_normal_norm;
462 vec3<Real> ljj_p_1, lij_p_1, lij;
463 Real cot_at_j, cot_at_j_p_1;
464
465 for (Index j = 0; j<nn_number; ++j) {
466 //return j+1 element of ordered_nn_ids unless j has the last value then wrap around and return 0th element
467 j_p_1 = Neighbors<Index>::plus_one(j,nn_number);
468
469 lij = nodes_.nn_distances(node_id)[j];
470 lij_p_1 = nodes_.nn_distances(node_id)[j_p_1];
471 ljj_p_1 = lij_p_1 - lij;
472
473 cot_at_j = cot_between_vectors(lij, (-1)*ljj_p_1);
474 cot_at_j_p_1 = cot_between_vectors(lij_p_1, ljj_p_1);
475
476
477 face_normal = lij.cross(lij_p_1); //nodes_.nn_distances(node_id)[j].cross(nodes_.nn_distances(node_id)[j_p_1]);
478 face_normal_norm = face_normal.norm();
479 face_area = mixed_area(lij, lij_p_1, Real(0.5)*face_normal_norm, cot_at_j, cot_at_j_p_1);
480 area_sum += face_area;
481 face_normal_sum += face_area*face_normal/face_normal_norm;
482
483 local_curvature_vec -= (cot_at_j_p_1*lij + cot_at_j*lij_p_1);
484 }
485
486 nodes_.set_area(node_id, area_sum);
487 nodes_.set_volume(node_id, nodes_[node_id].pos.dot(face_normal_sum)/((Real) 3.)); // 18=3*6: 6 has the aforementioned justification. 3 is part of the formula for the tetrahedron volume
488 nodes_.set_curvature_vec(node_id, -local_curvature_vec/((Real) 2.*area_sum)); // 2 is part of the formula to calculate the local curvature I just did not divide the vector inside the loop
489 nodes_.set_unit_bending_energy(node_id, local_curvature_vec.dot(local_curvature_vec)/((Real) 8.*area_sum)); // 8 is 2*4, where 4 is the square of the above two and the area in the denominator is what remains after canceling. 1/ comes from the pre-factor to bending energy
490
491 };

494 static std::tuple<Real, vec3<Real>> partial_voronoi_area_and_face_normal_of_node_in_a_triangle(vec3<Real> const& lij,
495 vec3<Real> const& lij_p_1)
496 {
502 Real area, face_normal_norm;
503 vec3<Real> un_noremd_face_normal;
504 //precalculating this normal and its norm, will be needed in area calc. If all triangles are oriented as
505 // right-handed, then this normal will point outwards
506 un_noremd_face_normal = lij.cross(lij_p_1);
507 face_normal_norm = un_noremd_face_normal.norm();
508 area = mixed_area(lij, lij_p_1, face_normal_norm/2.);
509 return std::make_tuple(area, un_noremd_face_normal);

512 static Real mixed_area(vec3<Real> const& lij, vec3<Real> const& lij_p_1, Real triangle_area, Real cot_at_j, Real cot_at_j_p_1){
513 if ((cot_at_j>0.) && (cot_at_j_p_1>0.)) { // both angles at j and j+1 are smaller than 90 deg so the triangle can only be obtuse at the node
514 if (lij.dot(lij_p_1)>0) { // cos at i is positive i.e. angle at i is not obtuse
515 return (cot_at_j_p_1*lij.dot(lij) + cot_at_j*lij_p_1.dot(lij_p_1))/Real(8.);
516 }
517 else {//obtuse at node i.
518 return triangle_area/Real(2.);
519 }
520 }
521 else {//obtuse at node j or j+1.
522 return triangle_area/Real(4.);
523 }
524
525 }
